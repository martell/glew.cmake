cmake_minimum_required(VERSION 2.8.10)

set(THIS_LIBRARY glew)
project( ${THIS_LIBRARY} )

if(MINGW OR UNIX)
  set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -Wno-attributes")
endif()

set(GLEW_SOURCE_DIR ${CMAKE_CURRENT_SOURCE_DIR}/glew)

include_directories( ${GLEW_SOURCE_DIR}/include)

add_definitions( "-DGLEW_STATIC" )

file( GLOB GLEW_HEADERS ${GLEW_SOURCE_DIR}/include/GL/*.h )
file( GLOB GLEW_SOURCES ${GLEW_SOURCE_DIR}/src/glew.c )

add_library( ${THIS_LIBRARY} STATIC ${GLEW_SOURCES} ${GLEW_HEADERS} )



set_target_properties(${THIS_LIBRARY} PROPERTIES PUBLIC_HEADER GLEW_HEADERS)

set(BINDIR bin)
set(BINCOMPONENT runtime)
set(LIBDIR lib)
set(LIBCOMPONENT dev)
set(INCLUDEDIR include)
set(INCLUDECOMPONENT dev)

install(TARGETS ${THIS_LIBRARY}
	RUNTIME DESTINATION ${BINDIR} COMPONENT ${BINCOMPONENT}
	LIBRARY DESTINATION ${LIBDIR} COMPONENT ${LIBCOMPONENT}
	ARCHIVE DESTINATION ${LIBDIR} COMPONENT ${LIBCOMPONENT}
	PUBLIC_HEADER DESTINATION ${INCLUDEDIR} COMPONENT ${INCLUDECOMPONENT})
